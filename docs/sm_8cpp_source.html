111  void StateMachine::_swl_execute_switching_logic(util::control* signals_rc){
112 
113  if ((mSW_CONTROL_ACTIVE && (signals_rc->velocity < PWM_CLEAR_TH_LOW))
114  ||!(mInterruptManager->mInterruptGroups[1]->mInterrupts.front()->mStatus)
115  #ifdef ROS_ACTIVE
116  || mRosi->mIsIdle
117  #endif /* ROS_ACTIVE */
118  ){
119  mSWLState = SWStates::OFF;
120  mSW_CONTROL_ACTIVE = false;
121  mStatusIndicator->switch_status(StatusIndicator::SIStatus::RC_ON);
122  }else{;}
123 
124 
125  if (util::HAS_CLEAR_STATE(signals_rc->differential_front) && util::HAS_CLEAR_STATE(signals_rc->differential_rear)){
126  mS1 = util::STATE(signals_rc->differential_front);
127  mS2 = util::STATE(signals_rc->differential_rear);
128 
129  util::print("mS1: ", false);
130  util::print(mS1, true);
131  util::print("mS2: ", false);
132  util::print(mS2, true);
133  util::print("dt: ", false);
134  util::print(float(us2ms*(micros()-mSWLTimestamp)), true);
135 
136  switch(mSWLState){
137  case SWStates::OFF: {
138  mSWLTimestamp = micros();
139  if (mS1^mS2) {
140  mS1_prev = mS1;
141  mS2_prev = mS2;
142  mSWLState = SWStates::ARM;
143  }else{;}
144  break;
145  }
146  case SWStates::ARM: {
147  if ((mS1^mS2)&&(mS1!=mS1_prev)&&(mS2!=mS2_prev)) {
148  mS1_prev = mS1;
149  mS2_prev = mS2;
150  mSWLState = SWStates::SWITCH;
151  }
152  else if(!(mS1^mS2)){
153  _swl_check_idle_transition();
154  }
155  else{
156  mSWLTimestamp = micros();
157  }
158  break;
159  }
160  case SWStates::SWITCH: {
161  _swl_check_idle_transition();
162  if ((mS1^mS2)&&(mS1!=mS1_prev)&&(mS2!=mS2_prev)) {
163  mS1_prev = mS1;
164  mS2_prev = mS2;
165  mSWLState = SWStates::OFF;
166  mSW_CONTROL_ACTIVE = true;
167  mStatusIndicator->switch_status(StatusIndicator::SIStatus::SW_ON);
168  }else{;}
169  break;
170  }
171  default: ;
172  }
173  } else{;}
174 
175  util::print("SW_CONTROL_ACTIVE:",false);
176  util::print(mSW_CONTROL_ACTIVE, true);
177  util::print("STATEMACHINE STATE:",false);
178  util::print(mSWLState, true);
179  }
